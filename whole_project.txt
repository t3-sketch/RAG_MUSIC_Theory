 1 | """Claude API を呼び出して音楽理論解説を生成するモジュール。"""
 2 | from anthropic import Anthropic
 3 | 
 4 | import config
 5 | 
 6 | SYSTEM_PROMPT = """あなたは音楽理論と楽曲分析の専門家です。
 7 | 以下の参考資料（音楽理論教材からの抜粋）を根拠に、ユーザーの質問へ日本語で丁寧かつ具体的に答えてください。
 8 | 
 9 | ルール:
10 | - 参考資料に書かれている内容を優先して使うこと。
11 | - 参考資料に無い内容を補う場合は「(資料外の一般知識)」と明示すること。
12 | - 楽曲の音響特徴が与えられている場合は、それを資料の理論と結びつけて解説すること。
13 | - コード進行やキーに言及するときは、機能（トニック/サブドミナント/ドミナント等）にも触れること。
14 | 
15 | [参考資料]
16 | {context}
17 | """
18 | 
19 | 
20 | def explain(query: str, chunks: list[dict], audio_desc: str | None = None) -> str:
21 |     """検索したチャンクと（あれば）音響特徴を元に解説を生成する。"""
22 |     client = Anthropic(api_key=config.ANTHROPIC_API_KEY)
23 | 
24 |     context = "\n\n---\n\n".join(
25 |         f"[出典: {c['meta'].get('source', '?')}]\n{c['text']}" for c in chunks
26 |     ) or "(参考資料が見つかりませんでした)"
27 | 
28 |     parts = []
29 |     if audio_desc:
30 |         parts.append(f"# 解析した楽曲の音響特徴\n{audio_desc}")
31 |     parts.append(f"# 質問\n{query}")
32 |     user_content = "\n\n".join(parts)
33 | 
34 |     resp = client.messages.create(
35 |         model=config.CLAUDE_MODEL,
36 |         max_tokens=config.MAX_TOKENS,
37 |         system=SYSTEM_PROMPT.format(context=context),
38 |         messages=[{"role": "user", "content": user_content}],
39 |     )
40 |     return "".join(b.text for b in resp.content if b.type == "text")

/README.md:
--------------------------------------------------------------------------------
  1 | # 🎵 music-rag — RAG 楽曲解説システム
  2 | 
  3 | 音楽理論教材を知識ベースにした **RAG（検索拡張生成）** で、楽曲についての質問に音楽理論の観点から答えるローカルアプリ。
  4 | ローカル音源（Librosa）または web 上の楽曲（Songle API）の音響特徴を解析し、教材の理論と結びつけて Claude が日本語で解説する。
  5 | 
  6 | > 個人利用を前提とした最小構成。Python ワンスタック（Streamlit）で、サーバー不要。
  7 | 
  8 | ---
  9 | 
 10 | ## ✨ 特徴
 11 | 
 12 | - **教材ベースの解説** — 投入した教材（SoundQuest / musicplanz など）を根拠に回答。資料外の知識は明示。
 13 | - **2系統の音響解析**
 14 |   - ローカルファイル（mp3/wav/flac）→ Librosa で BPM・キー・音色を推定
 15 |   - web 上の楽曲 URL → Songle API（産総研）で **実際のコード進行・ビート・サビ構造** を取得
 16 | - **透明性** — 回答に使った教材チャンクの出典と、解析した音響特徴を画面で確認できる。
 17 | - **ローカル完結** — Chroma DB はローカル永続化。外部公開なし。
 18 | 
 19 | ---
 20 | 
 21 | ## 🧭 パイプライン
 22 | 
 23 | ```
 24 | 質問テキスト ─┐
 25 |               ├─→ クエリ統合 → BGE-M3 埋め込み → Chroma 検索(top-k) ─┐
 26 | 音源(任意) ───┘                                                      ├─→ Claude API → 解説テキスト
 27 |    ├ ローカル → Librosa(BPM/Key/音色)                                │
 28 |    └ URL     → Songle(コード進行/ビート/構造) ──→ 音響特徴をテキスト化 ┘
 29 |                                                   教材コーパス(事前にChromaへ投入)
 30 | ```

  1 | """音響解析モジュール。
  2 | 
  3 | 2系統の入力をサポートする:
  4 |   1. ローカル音声ファイル（mp3/wav/flac）  -> Librosa で特徴抽出
  5 |   2. web上の楽曲URL（YouTube/ニコニコ等）  -> Songle API で解析結果を取得
  6 | 
  7 | どちらも最終的に「Claude に渡す日本語テキスト」へ変換する。
  8 | """
  9 | from __future__ import annotations
 10 | 
 11 | import numpy as np
 12 | import requests
 13 | 
 14 | import config
 15 | 
 16 | # --- 音名・調推定用プロファイル（Krumhansl-Schmuckler）---
 17 | PITCHES = ["C", "C#", "D", "D#", "E", "F", "F#", "G", "G#", "A", "A#", "B"]
 18 | _KS_MAJOR = np.array(
 19 |     [6.35, 2.23, 3.48, 2.33, 4.38, 4.09, 2.52, 5.19, 2.39, 3.66, 2.29, 2.88]
 20 | )
 21 | _KS_MINOR = np.array(
 22 |     [6.33, 2.68, 3.52, 5.38, 2.60, 3.53, 2.54, 4.75, 3.98, 2.69, 3.34, 3.17]
 23 | )
 24 | 
 25 | 
 26 | # =========================================================
 27 | #  1. ローカルファイル: Librosa
 28 | # =========================================================
 29 | def extract_local(path: str) -> dict:
 30 |     """ローカル音声ファイルから音響特徴を抽出する。"""
 31 |     import librosa  # 重いので関数内 import
 32 | 
 33 |     y, sr = librosa.load(path, sr=22050, mono=True)
 34 | 
 35 |     tempo, _ = librosa.beat.beat_track(y=y, sr=sr)
 36 |     tempo = float(np.atleast_1d(tempo)[0])
 37 | 
 38 |     chroma = librosa.feature.chroma_cens(y=y, sr=sr).mean(axis=1)
 39 |     key = _estimate_key(chroma)
 40 | 
 41 |     centroid = float(librosa.feature.spectral_centroid(y=y, sr=sr).mean())
 42 |     rms = float(librosa.feature.rms(y=y).mean())
 43 |     zcr = float(librosa.feature.zero_crossing_rate(y).mean())
 44 | 
 45 |     return {
 46 |         "bpm": round(tempo, 1),
 47 |         "key": key,
 48 |         "brightness_hz": round(centroid, 0),
 49 |         "energy_rms": round(rms, 4),
 50 |         "zcr": round(zcr, 4),
 51 |         "duration_sec": round(len(y) / sr, 1),
 52 |     }
 53 | 
 54 | 
 55 | def _estimate_key(chroma_mean: np.ndarray) -> str:
 56 |     """12次元クロマから最も相関の高い調（major/minor）を返す。"""
 57 |     best_name, best_score = "不明", -2.0
 58 |     for i in range(12):
 59 |         for profile, mode in ((_KS_MAJOR, "major"), (_KS_MINOR, "minor")):
 60 |             rolled = np.roll(profile, i)
 61 |             score = float(np.corrcoef(rolled, chroma_mean)[0, 1])
 62 |             if score > best_score:
 63 |                 best_score = score
 64 |                 best_name = f"{PITCHES[i]} {mode}"
 65 |     return best_name
 66 | 
 67 | 
 68 | def describe_local(f: dict) -> str:
 69 |     """Librosa の特徴を日本語の説明テキストにする。"""
 70 |     brightness = "明るい" if f["brightness_hz"] > 2500 else "落ち着いた"
 71 |     energy = "強い" if f["energy_rms"] > 0.05 else "穏やか"
 72 |     return (
 73 |         f"- 解析方法: Librosa（ローカル音源の信号解析）\n"
 74 |         f"- 推定BPM: {f['bpm']}\n"

 87 | def fetch_songle(url: str) -> dict:

 94 |     headers = {}

 96 |         # 認証方式はダッシュボードで要確認（ヘッダ or クエリ）。

 98 | 
 99 |     def _get(endpoint: str):
100 |         r = requests.get(
101 |             f"{config.SONGLE_API_BASE}/{endpoint}",
102 |             params=params,
103 |             headers=headers,
104 |             timeout=20,
105 |         )

109 |     song = _get("song")

125 | def _list(data, key: str) -> list:
126 |     """レスポンスの形ゆれを吸収して目的のリストを取り出す。"""
127 |     if isinstance(data, list):
128 |         return data
129 |     if isinstance(data, dict):
130 |         if key in data and isinstance(data[key], list):
131 |             return data[key]
132 |         for v in data.values():
133 |             if isinstance(v, dict) and key in v and isinstance(v[key], list):
134 |                 return v[key]
135 |             if isinstance(v, list):
136 |                 return v
137 |     return []

140 | def _dig(data, *keys):
141 |     cur = data
142 |     for k in keys:
143 |         if not isinstance(cur, dict):
144 |             return None
145 |         cur = cur.get(k)
146 |     return cur

149 | def _bpm_from_beats(beats: list) -> float | None:
150 |     """連続する拍の間隔の中央値から BPM を推定する。"""
151 |     starts = [b.get("start") for b in beats if isinstance(b, dict) and "start" in b]
152 |     starts = sorted(s for s in starts if s is not None)
153 |     if len(starts) < 2:
154 |         return None
155 |     diffs = np.diff(starts)
156 |     median = float(np.median(diffs))
157 |     if median <= 0:
158 |         return None
159 |     # start がミリ秒で来る場合（間隔が大きい）は秒に換算
160 |     interval_sec = median / 1000 if median > 10 else median
161 |     return round(60.0 / interval_sec, 1)

 95 | def _parse_content_blocks(content, page_url: str) -> list[dict]:
 96 |     """本文 DOM を走査し、テキスト・音声・画像を種別付きで抽出する。"""
 97 |     entries: list[dict] = []
 98 | 
 99 |     for elem in content.children:
100 |         if not hasattr(elem, "name") or not elem.name:
101 |             continue
102 | 
103 |         # ── テキストブロック（p, h2-h5, dl, blockquote, ul, ol など）──
104 |         if elem.name in {"p", "h2", "h3", "h4", "h5", "dl", "blockquote", "ul", "ol"}:
105 |             text = _clean_text(elem.get_text(separator=" "))
106 |             if text:
107 |                 entries.append({"text": text, "type": "text", "source_url": page_url})
108 | 
109 |         # ── .image ブロック（画像 + 直後の音声プレーヤーを一体で処理）──
110 |         elif "image" in elem.get("class", []):
111 |             block_text = _extract_image_block(elem)
112 |             if block_text:
113 |                 entries.append({"text": block_text, "type": "image", "source_url": page_url})
114 | 
115 |         # ── 単体の audio（.image の外に置かれているもの）──
116 |         elif elem.name == "audio" or elem.select("audio"):
117 |             audio_text = _extract_audio_text(elem, surrounding="")
118 |             if audio_text:
119 |                 entries.append({"text": audio_text, "type": "audio", "source_url": page_url})
120 | 
121 |         # ── Spotify 埋め込み ──
122 |         elif elem.name == "div" and "sp-playlist" in elem.get("class", []):
123 |             iframe = elem.select_one("iframe[src*='open.spotify.com']")
124 |             if iframe:
125 |                 src = iframe.get("src", "")
126 |                 entries.append({
127 |                     "text": f"[Spotify プレイリスト埋め込み] URL: {src}",
128 |                     "type": "text",
129 |                     "source_url": page_url,
130 |                 })
131 | 
132 |         # ── .def（用語定義ボックス）──
133 |         elif elem.name == "dl" and "def" in elem.get("class", []):
134 |             text = _clean_text(elem.get_text(separator=" "))
135 |             if text:
136 |                 entries.append({"text": f"[用語定義] {text}", "type": "text", "source_url": page_url})
137 | 
138 |     return entries

141 | def _extract_image_block(elem) -> str:
142 |     """
143 |     .image ブロックから、画像の alt テキストと
144 |     直後に埋め込まれた <audio> の MP3 URL を合成してテキスト化する。
145 |     """
146 |     parts: list[str] = []
147 | 
148 |     # imgcaption
149 |     caption_el = elem.select_one(".imgcaption")
150 |     if caption_el:
151 |         cap = _clean_text(caption_el.get_text())
152 |         if cap:
153 |             parts.append(f"[図のキャプション] {cap}")
154 | 
155 |     # img alt
156 |     for img in elem.select("img"):
157 |         alt = img.get("alt", "").strip()
158 |         src = img.get("src", "")
159 |         if alt:
160 |             parts.append(f"[画像の説明] {alt}")
161 |         if src:
162 |             parts.append(f"[画像URL] {src}")
163 | 
164 |     # 直後にある audio → MP3 URL + 周辺テキスト
165 |     audio = elem.select_one("audio.wp-audio-shortcode")
166 |     if audio:
167 |         surrounding = " ".join(p.strip() for p in parts)
168 |         parts.append(_extract_audio_text(audio, surrounding))
169 | 
170 |     return "\n".join(filter(None, parts))

173 | def _extract_audio_text(audio_elem, surrounding: str) -> str:
174 |     """audio 要素から MP3 URL を取り出し、周辺テキストと合成する。"""
175 |     source = audio_elem.select_one("source[type='audio/mpeg']")
176 |     if not source:
177 |         return ""
178 |     mp3_url = source.get("src", "").split("?")[0]  # クエリパラメータを除去
179 |     if not mp3_url:
180 |         return ""
181 |     ctx = f" （周辺文脈: {surrounding[:120]}）" if surrounding else ""
182 |     return f"[音声サンプル] MP3: {mp3_url}{ctx}"

212 | # ── チャンク分割 ──────────────────────────────────────────────────────────────
213 | 
214 | def chunk_text(text: str) -> list[str]:
215 |     """改行を優先しつつ、文字数ベースでオーバーラップ付きに分割する。"""
216 |     import re
217 |     text = re.sub(r"\n{3,}", "\n\n", text).strip()
218 |     chunks, start = [], 0
219 |     n = len(text)
220 |     while start < n:
221 |         end = min(start + config.CHUNK_CHARS, n)
222 |         if end < n:
223 |             window = text[start:end]
224 |             for sep in ("\n\n", "\n", "。", ". "):
225 |                 idx = window.rfind(sep)
226 |                 if idx > config.CHUNK_CHARS * 0.4:
227 |                     end = start + idx + len(sep)
228 |                     break
229 |         chunk = text[start:end].strip()
230 |         if chunk:
231 |             chunks.append(chunk)
232 |         start = max(end - config.CHUNK_OVERLAP, end if end > start else start + 1)
233 |     return chunks

238 | def upsert_chunks(col, source_id: str, chunks: list[str], base_meta: dict) -> int:
239 |     """チャンクを Chroma に upsert する。投入数を返す。"""
240 |     if not chunks:
241 |         return 0
242 |     ids = [f"{source_id}::{i}" for i in range(len(chunks))]
243 |     metas = [{**base_meta, "chunk_index": i} for i in range(len(chunks))]
244 |     col.upsert(ids=ids, documents=chunks, metadatas=metas)
245 |     return len(chunks)
